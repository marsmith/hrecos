Process for importing legacy HRECOS CSV tables into mysql/mariadb database
------------------------------------------------------------------------------

1.  Create new database
    CREATE DATABASE hrecos;

2.  Create new table for each csv ('raw' and 'post') and load data

    CREATE TABLE IF NOT EXISTS hrecos.raw (`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, site_name VARCHAR(12), `parameter` VARCHAR(12), `datetime` DATETIME, value FLOAT, flag VARCHAR(12), comment VARCHAR(12), tz_code VARCHAR(12), agency_id VARCHAR(12));

3. Import Data
    mysqlimport -h localhost -P 3306 -u hawdis -pabc123 --local --ignore-lines=1 --fields-terminated-by=',' --lines-terminated-by='\n' hrecos raw.csv -v

4. Create indexes
    CREATE INDEX raw_datetime_IDX ON hrecos.raw (`datetime`);
    CREATE INDEX raw_site_name_IDX ON hrecos.raw (`site_name`);
    CREATE INDEX raw_parameter_IDX ON hrecos.raw (`parameter`);

Remote import (Final):
1) CREATE TABLE IF NOT EXISTS hrecos.post (site_name TEXT, `parameter` TEXT, `datetime` DATETIME, value FLOAT, flag TEXT, comment TEXT, tz_code TEXT, agency_id TEXT);
2) mysqlimport -h mysql.natweb.usgs.gov -P 3306 -u hrecos -prsT13zKDN9fp9EVI --local --ignore-lines=1 --fields-terminated-by=',' --lines-terminated-by='\n' hrecos 3) post.csv -v
